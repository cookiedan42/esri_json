16/// Representation Symbol Layers usable in a a [`PolygonSymbol3D`]
17#[derive(Clone, Serialize, Deserialize, Debug, PartialEq)]
18#[serde(untagged)]
19pub enum PolygonSymbol3DLayers {
20    Extrude(ExtrudeSymbol3DLayer),
21    Fill(FillSymbol3DLayer),
22    Icon(IconSymbol3DLayer),
23    #[deprecated = "use [`fillSymbol3DLayer`] with outline instead"]
24    Line(LineSymbol3DLayer),
25    Object(ObjectSymbol3DLayer),
26    #[deprecated = "use [`labelSymbol3D`] for TextSymbol3dLayers"]
27    Text(TextSymbol3DLayer),
28    Water(WaterSymbol3DLayer),
29}
30

71/// Representation of a [PolygonSymbol3D](https://developers.arcgis.com/web-scene-specification/objects/polygonSymbol3D/)
72#[derive(Clone, Serialize, Deserialize, Debug, PartialEq, Default)]
73pub struct PolygonSymbol3D {
74    #[serde(rename = "type")]
75    _type: T,
76
77    #[serde(skip_serializing_if = "Option::is_none", rename = "symbolLayers")]
78    symbol_layers: Option<Vec<PolygonSymbol3DLayers>>,
79    #[serde(skip_serializing_if = "Option::is_none", rename = "styleOrigin")]
80    style_origin: Option<StyleOrigin>,
81}
82
83impl<Layer> From<Layer> for PolygonSymbol3D
84where
85    Layer: Into<PolygonSymbol3DLayers>,
86{
87    fn from(val: Layer) -> Self {
88        Self {
89            _type: T,
90            symbol_layers: Some(vec![val.into()]),
91            style_origin: None,
92        }
93    }
94}
95
96///Builder pattern
97impl PolygonSymbol3D {
98    /// A Collection of Symbol3DLayer objects used to visualize the graphic or feature.
99    pub fn symbol_layers(mut self, symbol_layers: Vec<PolygonSymbol3DLayers>) -> Self {
100        self.symbol_layers = Some(symbol_layers);
101        self
102    }
103    /// The origin of the style from which the symbol was originally referenced.
104    pub fn style_origin(mut self, style_origin: StyleOrigin) -> Self {
105        self.style_origin = Some(style_origin);
106        self
107    }
