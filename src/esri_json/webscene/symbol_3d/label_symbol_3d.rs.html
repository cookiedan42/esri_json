1//! Representation of a [LabelSymbol3D](https://developers.arcgis.com/web-scene-specification/objects/labelSymbol3D/)
2//!
3//! LabelSymbol3D is used to render labels for features from a FeatureLayer in a 3D SceneView.
4
5use crate::common::marker_type;
6use crate::webscene::text_symbol_3d_layer::TextSymbol3DLayer;
7use serde::{Deserialize, Serialize};
8
9pub use crate::webscene::common::{Callout, VerticalOffset};
10
11/// Representation Symbol Layers usable in a a [`LabelSymbol3D`]
12#[derive(Clone, Serialize, Deserialize, Debug, PartialEq)]
13#[serde(untagged)]
14pub enum LabelSymbol3DLayers {
15    Text(TextSymbol3DLayer),
16}
17impl From<TextSymbol3DLayer> for LabelSymbol3DLayers {
18    fn from(val: TextSymbol3DLayer) -> Self {
19        LabelSymbol3DLayers::Text(val)
20    }
21}
22
23marker_type!(T, "LabelSymbol3D");
24
25/// Representation of a [LabelSymbol3D](https://developers.arcgis.com/web-scene-specification/objects/labelSymbol3D/)
26#[derive(Clone, Serialize, Deserialize, Debug, PartialEq)]
27pub struct LabelSymbol3D {
28    #[serde(rename = "type")]
29    _type: T,
30
31    #[serde(skip_serializing_if = "Option::is_none")]
32    callout: Option<Callout>,
33    #[serde(skip_serializing_if = "Option::is_none", rename = "symbolLayers")]
34    symbol_layers: Option<Vec<LabelSymbol3DLayers>>,
35    #[serde(skip_serializing_if = "Option::is_none", rename = "verticalOffset")]
36    vertical_offset: Option<VerticalOffset>,
37}
38impl<Layer> From<Layer> for LabelSymbol3D
39where
40    Layer: Into<LabelSymbol3DLayers>,
41{
42    fn from(val: Layer) -> Self {
43        Self {
44            _type: T,
45            callout: None,
46            symbol_layers: Some(vec![val.into()]),
47            vertical_offset: None,
48        }
49    }
50}
51
52/// Builder pattern
53impl LabelSymbol3D {
54    pub fn callout(mut self, callout: Callout) -> Self {
55        self.callout = Some(callout);
56        self
57    }
58    pub fn symbol_layers(mut self, symbol_layers: Vec<LabelSymbol3DLayers>) -> Self {
59        self.symbol_layers = Some(symbol_layers);
60        self
61    }
62    pub fn vertical_offset(mut self, vertical_offset: VerticalOffset) -> Self {
63        self.vertical_offset = Some(vertical_offset);
64        self
65    }
66}

72    #[test]
73    fn example() {
74        let label_symbol_3d = r#"{
75            "type": "LabelSymbol3D",
76            "symbolLayers": [
77                {
78                "size": 12,
79                "type": "Text",
80                "material": {
81                    "color": [
82                    255,
83                    255,
84                    0
85                    ]
86                }
87                }
88            ]
89        }"#;
90        let de: LabelSymbol3D = serde_json::from_str(label_symbol_3d).unwrap();
91        let ser = serde_json::to_string(&de).unwrap();
92        let serde: LabelSymbol3D = serde_json::from_str(&ser).unwrap();
93        assert_eq!(serde, de);
