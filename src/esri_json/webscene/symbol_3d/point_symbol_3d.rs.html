1//! Representation of a [PointSymbol3D](https://developers.arcgis.com/web-scene-specification/objects/pointSymbol3D/)
2//!
3//! PointSymbol3D is used to render features with Point geometry in a 3D SceneView.
4
5use crate::common::marker_type;
6use crate::webscene::common::{Callout, StyleOrigin, VerticalOffset};
7use crate::webscene::{
8    icon_symbol_3d_layer::IconSymbol3DLayer, object_symbol_3d_layer::ObjectSymbol3DLayer,
9    text_symbol_3d_layer::TextSymbol3DLayer,
10};
11use serde::{Deserialize, Serialize};
12
13/// Representation Symbol Layers usable in a a [`PointSymbol3D`]
14#[derive(Clone, Serialize, Deserialize, Debug, PartialEq)]
15#[serde(untagged)]
16pub enum PointSymbol3DLayers {
17    Icon(IconSymbol3DLayer),
18    Object(ObjectSymbol3DLayer),
19    Text(TextSymbol3DLayer),
20}
21impl From<IconSymbol3DLayer> for PointSymbol3DLayers {
22    fn from(val: IconSymbol3DLayer) -> Self {
23        PointSymbol3DLayers::Icon(val)
24    }
25}
26impl From<ObjectSymbol3DLayer> for PointSymbol3DLayers {
27    fn from(val: ObjectSymbol3DLayer) -> Self {
28        PointSymbol3DLayers::Object(val)
29    }
30}
31impl From<TextSymbol3DLayer> for PointSymbol3DLayers {
32    fn from(val: TextSymbol3DLayer) -> Self {
33        PointSymbol3DLayers::Text(val)
34    }
35}
36
37marker_type!(T, "PointSymbol3D");
38
39/// Representation of a [PointSymbol3D](https://developers.arcgis.com/web-scene-specification/objects/pointSymbol3D/)
40#[derive(Clone, Serialize, Deserialize, Debug, PartialEq)]
41pub struct PointSymbol3D {
42    #[serde(rename = "type")]
43    _type: T,
44
45    #[serde(skip_serializing_if = "Option::is_none")]
46    callout: Option<Callout>,
47    #[serde(skip_serializing_if = "Option::is_none", rename = "styleOrigin")]
48    style_origin: Option<StyleOrigin>,
49    #[serde(skip_serializing_if = "Option::is_none", rename = "symbolLayers")]
50    symbol_layers: Option<Vec<PointSymbol3DLayers>>,
51    #[serde(skip_serializing_if = "Option::is_none", rename = "verticalOffset")]
52    vertical_offset: Option<VerticalOffset>,
53}
54impl<Layer> From<Layer> for PointSymbol3D
55where
56    Layer: Into<PointSymbol3DLayers>,
57{
58    fn from(val: Layer) -> Self {
59        Self {
60            _type: T,
61            callout: None,
62            style_origin: None,
63            symbol_layers: Some(vec![val.into()]),
64            vertical_offset: None,
65        }
66    }
67}
68
69/// Builder pattern
70impl PointSymbol3D {
71    /// Callout configuration for a symbol.
72    pub fn callout(mut self, callout: Callout) -> Self {
73        self.callout = Some(callout);
74        self
75    }
76    /// The origin of the style from which the symbol was originally referenced.
77    pub fn style_origin(mut self, style_origin: StyleOrigin) -> Self {
78        self.style_origin = Some(style_origin);
79        self
80    }
81    /// A Collection of Symbol3DLayer objects used to visualize the graphic or feature.
82    pub fn symbol_layers(mut self, symbol_layers: Vec<PointSymbol3DLayers>) -> Self {
83        self.symbol_layers = Some(symbol_layers);
84        self
85    }
86    /// Shifts the symbol along the vertical world axis by a given length.
87    pub fn vertical_offset(mut self, vertical_offset: VerticalOffset) -> Self {
88        self.vertical_offset = Some(vertical_offset);
89        self
90    }
91}

97    #[test]
98    fn point_symbol_3d_example() {
99        let point_symbol_3d = r#"{
100            "type": "PointSymbol3D",
101            "symbolLayers": [
102                {
103                "width": 30000,
104                "height": 160000,
105                "resource": {
106                    "primitive": "cylinder"
107                },
108                "type": "Object",
109                "material": {
110                    "color": [
111                    0,
112                    255,
113                    0
114                    ]
115                }
116                }
117            ]
118        }"#;
119        let de: PointSymbol3D = serde_json::from_str(point_symbol_3d).unwrap();
120        let ser = serde_json::to_string(&de).unwrap();
121        let serde: PointSymbol3D = serde_json::from_str(&ser).unwrap();
122        assert_eq!(serde, de);
123    }
