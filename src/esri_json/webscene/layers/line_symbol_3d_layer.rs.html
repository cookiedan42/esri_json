1//! Representation of a [LineSymbol3DLayer](https://developers.arcgis.com/web-scene-specification/objects/lineSymbol3DLayer/)  
2//!
3//! LineSymbol3DLayer renders Polyline geometries using a flat 2D line with a LineSymbol3D in a 3D SceneView.
4//!
5//! Includes re-exports of relevant types
6//!
7
8use crate::common::marker_type;
9use crate::webscene::common::LinePatternInternal;
10use serde::{Deserialize, Serialize};
11
12pub use crate::common::Color;
13pub use crate::webscene::common::line_marker::{LineMarker, LineSymbolMarkerStyle, Placement};
14pub use crate::webscene::common::{LineCap, LineJoin, LinePattern, Material, PxOrPt};
15
16marker_type!(T, "Line");
17
18/// Representation of a [LineSymbol3DLayer](https://developers.arcgis.com/web-scene-specification/objects/lineSymbol3DLayer/)  
19#[derive(Clone, Serialize, Deserialize, Debug, PartialEq, Default)]
20pub struct LineSymbol3DLayer {
21    #[serde(rename = "type")]
22    _type: T,
23
24    #[serde(skip_serializing_if = "Option::is_none")]
25    cap: Option<LineCap>,
26    #[serde(skip_serializing_if = "Option::is_none")]
27    join: Option<LineJoin>,
28    #[serde(skip_serializing_if = "Option::is_none")]
29    marker: Option<LineMarker>,
30    #[serde(skip_serializing_if = "Option::is_none")]
31    material: Option<Material>,
32    #[serde(skip_serializing_if = "Option::is_none")]
33    pattern: Option<LinePatternInternal>,
34    #[serde(skip_serializing_if = "Option::is_none")]
35    size: Option<PxOrPt>,
36}
37
38/// Builder pattern
39impl LineSymbol3DLayer {
40    /// Shape of the tips at the start and end of each line geometry. This also applies to the tips of each pattern segment along the line
41    pub fn cap(mut self, cap: LineCap) -> Self {
42        self.cap = Some(cap);
43        self
44    }
45    /// Shape of the intersection of two line segments
46    pub fn join(mut self, join: LineJoin) -> Self {
47        self.join = Some(join);
48        self
49    }
50    /// Represents markers placed at the start and end of each line geometry, or both
51    pub fn marker(mut self, marker: LineMarker) -> Self {
52        self.marker = Some(marker);
53        self
54    }
55    /// The material used to shade the geometry
56    pub fn material(mut self, material: Material) -> Self {
57        self.material = Some(material);
58        self
59    }
60    /// A pattern used to render a line
61    pub fn pattern(mut self, pattern: LinePattern) -> Self {
62        self.pattern = Some(pattern.into());
63        self
64    }
65    /// Line width in points, positive only
66    pub fn size(mut self, size: PxOrPt) -> Self {
67        // Todo: enforce this constraint
68        self.size = Some(size);
69        self
70    }
71}
