14/// Representation of a [WaterSymbol3DLayer](https://developers.arcgis.com/web-scene-specification/objects/waterSymbol3DLayer/)
15///
16/// Symbol Layer that describes a water appearance on surfaces in a SceneView.
17#[derive(Clone, Serialize, Deserialize, Debug, PartialEq, Default)]
18pub struct WaterSymbol3DLayer {
19    #[serde(rename = "type")]
20    _type: T,
21    #[serde(skip_serializing_if = "Option::is_none")]
22    color: Option<Color>,
23    #[serde(skip_serializing_if = "Option::is_none", rename = "waterbodySize")]
24    waterbody_size: Option<WaterbodySize>,
25    #[serde(skip_serializing_if = "Option::is_none", rename = "waveDirection")]
26    wave_direction: Option<f64>,
27    #[serde(skip_serializing_if = "Option::is_none", rename = "waveStrength")]
28    wave_strength: Option<WaveStrength>,
29}
30
31impl WaterSymbol3DLayer {
32    pub fn color(mut self, color: Color) -> Self {
33        self.color = Some(color);
34        self
35    }
36
37    /// Size of the waterbody the symbol layer represents. Applications will display waves that are appropriate for the chosen body of water, for example ocean versus marina versus swimming pool.
38    pub fn waterbody_size(mut self, size: WaterbodySize) -> Self {
39        self.waterbody_size = Some(size);
40        self
41    }
42
43    /// Azimuthal bearing for direction of the waves. If ommitted, waves appear directionless. The value is interpreted as 'geographic' rotation, i.e. clockwise starting from north.
44    pub fn wave_direction(mut self, direction: f64) -> Self {
45        self.wave_direction = Some(direction);
46        self
47    }
48
49    /// The magnitude of the waves displayed on the waterbody. Strings roughly follow the [Douglas sea scale](https://en.wikipedia.org/wiki/Douglas_sea_scale), currently limited to lower degrees.
50    pub fn wave_strength(mut self, strength: WaveStrength) -> Self {
51        self.wave_strength = Some(strength);
52        self
53    }
54}
55
56/// Size of the waterbody the symbol layer represents. Applications will display waves that are appropriate for the chosen body of water, for example ocean versus marina versus swimming pool.
57#[derive(Clone, Serialize, Deserialize, Debug, PartialEq)]
58#[serde(rename_all = "camelCase")]
59pub enum WaterbodySize {
60    Small,
61    Medium,
62    Large,
63}
64
65/// The magnitude of the waves displayed on the waterbody. Strings roughly follow the [Douglas sea scale](https://en.wikipedia.org/wiki/Douglas_sea_scale), currently limited to lower degrees.
66#[derive(Clone, Serialize, Deserialize, Debug, PartialEq)]
67#[serde(rename_all = "camelCase")]
68pub enum WaveStrength {
69    Calm,
70    Rippled,
71    Slight,
72    Moderate,
73}

75#[cfg(test)]
76mod tests {
77    use super::*;
78
79    #[test]
80    fn defaults() {
81        let water_layer = r#"
82        {
83            "type": "Water"
84        }"#;
85
86        let de: WaterSymbol3DLayer = serde_json::from_str(water_layer).unwrap();
87        let ser = serde_json::to_string(&de).unwrap();
88        let serde: WaterSymbol3DLayer = serde_json::from_str(&ser).unwrap();
89        assert_eq!(serde, de);
90    }
91
