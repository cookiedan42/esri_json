1//! Representation of a [FillSymbol3DLayer](https://developers.arcgis.com/web-scene-specification/objects/fillSymbol3DLayer/)
2//!
3//! FillSymbol3DLayer is used to render the surfaces of flat 2D Polygon geometries and 3D volumetric meshes in a SceneView.
4//!
5//! Includes re-exports of relevant types
6//!
7
8use crate::common::marker_type;
9use crate::webscene::common::PolygonPatternInternal;
10use serde::{Deserialize, Serialize};
11
12pub use crate::common::Color;
13pub use crate::webscene::common::{
14    ColorMixMode, Edges, LineCap, LinePattern, MaterialColorMixMode as Material, Outline,
15    PolygonPattern,
16};
17
18marker_type!(T, "Fill");
19
20/// Representation of a [FillSymbol3DLayer](https://developers.arcgis.com/web-scene-specification/objects/fillSymbol3DLayer/)
21#[derive(Clone, Serialize, Deserialize, Debug, PartialEq, Default)]
22pub struct FillSymbol3DLayer {
23    #[serde(rename = "type")]
24    _type: T,
25    #[serde(skip_serializing_if = "Option::is_none", rename = "castShadows")]
26    cast_shadows: Option<bool>, // only used by mesh 3d
27    #[serde(skip_serializing_if = "Option::is_none")]
28    edges: Option<Edges>, // only used by mesh 3d
29    #[serde(skip_serializing_if = "Option::is_none")]
30    material: Option<Material>,
31    #[serde(skip_serializing_if = "Option::is_none")]
32    outline: Option<Outline>, // only used by polygon 3d
33    #[serde(skip_serializing_if = "Option::is_none")]
34    pattern: Option<PolygonPatternInternal>,
35}
36
37// Builder pattern
38impl FillSymbol3DLayer {
39    /// Only used by mesh 3d
40    pub fn cast_shadows(mut self, cast_shadows: bool) -> Self {
41        self.cast_shadows = Some(cast_shadows);
42        self
43    }
44    /// Specifies an edge visualization style (only applies to [`MeshSymbol3D`](crate::webscene::mesh_symbol_3d::MeshSymbol3DLayers)). Edges describe the style applied to visually important edges of 3D objects.
45    pub fn edges(mut self, edges: Edges) -> Self {
46        self.edges = Some(edges);
47        self
48    }
49    /// The material used to shade the geometry, including colorMixMode options.
50    pub fn material(mut self, material: Material) -> Self {
51        self.material = Some(material);
52        self
53    }
54    /// The outline of the symbol layer (only applies to [`PolygonSymbol3D`](crate::webscene::polygon_symbol_3d::PolygonSymbol3DLayers)).
55    pub fn outline(mut self, outline: Outline) -> Self {
56        self.outline = Some(outline);
57        self
58    }
59    /// The pattern used to render the fill of the polygon [`PolygonSymbol3D`](crate::webscene::polygon_symbol_3d::PolygonSymbol3DLayers)).
60    pub fn pattern(mut self, pattern: PolygonPattern) -> Self {
61        self.pattern = Some(pattern.into());
62        self
63    }
64}
