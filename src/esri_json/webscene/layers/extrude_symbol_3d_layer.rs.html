1//! Representation of a [ExtrudeSymbol3DLayer](https://developers.arcgis.com/web-scene-specification/objects/extrudeSymbol3DLayer/)
2//!
3//! ExtrudeSymbol3DLayer is used to render Polygon geometries by extruding them upward from the ground, creating a 3D volumetric object.
4//!
5//! Includes re-exports of relevant types
6//!
7
8use crate::common::marker_type;
9use serde::{Deserialize, Serialize};
10
11pub use crate::common::Color;
12pub use crate::webscene::common::{Edges, Material};
13
14marker_type!(TypeString, "Extrude");
15
16/// Representation of a [ExtrudeSymbol3DLayer](https://developers.arcgis.com/web-scene-specification/objects/extrudeSymbol3DLayer/)
17#[derive(Clone, Serialize, Deserialize, Debug, PartialEq, Default)]
18pub struct ExtrudeSymbol3DLayer {
19    #[serde(rename = "type")]
20    _type: TypeString,
21    #[serde(skip_serializing_if = "Option::is_none", rename = "castShadows")]
22    cast_shadows: Option<bool>,
23    #[serde(skip_serializing_if = "Option::is_none")]
24    edges: Option<Edges>,
25    #[serde(skip_serializing_if = "Option::is_none")]
26    material: Option<Material>,
27    #[serde(skip_serializing_if = "Option::is_none")]
28    size: Option<f64>,
29}
30
31/// Builder pattern
32impl ExtrudeSymbol3DLayer {
33    pub fn cast_shadows(mut self, cast_shadows: bool) -> Self {
34        self.cast_shadows = Some(cast_shadows);
35        self
36    }
37    pub fn edges(mut self, edges: Edges) -> Self {
38        self.edges = Some(edges);
39        self
40    }
41    pub fn material(mut self, material: Material) -> Self {
42        self.material = Some(material);
43        self
44    }
45    pub fn size(mut self, size: f64) -> Self {
46        self.size = Some(size);
47        self
48    }
49}
