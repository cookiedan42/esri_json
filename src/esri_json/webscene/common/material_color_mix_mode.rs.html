1//! Representation of a [MaterialColorMixMode](https://developers.arcgis.com/web-scene-specification/objects/materialColorMixMode/)
2//!
3//! The material used to shade the geometry, including colorMixMode options.
4
5use crate::common::Color;
6use serde::{Deserialize, Serialize};
7
8/// Representation of a [MaterialColorMixMode](https://developers.arcgis.com/web-scene-specification/objects/materialColorMixMode/)
9#[derive(Clone, Serialize, Deserialize, Debug, PartialEq, Default)]
10#[serde(rename_all = "camelCase")]
11pub struct MaterialColorMixMode {
12    #[serde(skip_serializing_if = "Option::is_none")]
13    color: Option<Color>,
14    #[serde(skip_serializing_if = "Option::is_none")]
15    color_mix_mode: Option<ColorMixMode>,
16    #[serde(skip_serializing_if = "Option::is_none")]
17    transparency: Option<f64>,
18}
19/// Builder pattern
20impl MaterialColorMixMode {
21    /// Color is represented as a three or four-element array.
22    pub fn color(mut self, color: Color) -> Self {
23        self.color = Some(color);
24        self
25    }
26    /// Controls how symbolLayer or visualVariable color is applied onto the underlying geometry color/texture.
27    /// This property only applies to [`FillSymbol3DLayer`](crate::webscene::layers::fill_symbol_3d_layer::FillSymbol3DLayer)
28    /// within [`MeshSymbol3D`](crate::webscene::mesh_symbol_3d::MeshSymbol3DLayers)).
29    pub fn color_mix_mode(mut self, color_mix_mode: ColorMixMode) -> Self {
30        self.color_mix_mode = Some(color_mix_mode);
31        self
32    }
33    /// A value between 100 (full transparency) and 0 (full opacity). Ignored if no color is specified.
34    pub fn transparency(mut self, transparency: f64) -> Self {
35        self.transparency = Some(transparency);
36        self
37    }
38}
39
40/// Controls how symbolLayer or visualVariable color is applied onto the underlying geometry color/texture.
41#[derive(Clone, Serialize, Deserialize, Debug, PartialEq)]
42#[serde(rename_all = "camelCase")]
43pub enum ColorMixMode {
44    Tint,
45    Replace,
46    Multiply,
47}
