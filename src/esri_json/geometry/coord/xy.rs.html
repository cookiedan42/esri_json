1use crate::geometry::Coord;
2use serde::{Deserialize, Serialize};

4/// Base Coordinate type with X and Y coordinates
5#[derive(Serialize, Deserialize, Debug, Clone, PartialEq)]
6#[serde(into = "Vec<f64>", from = "Vec<f64>")]
7pub struct CoordXy {
8    x: f64,
9    y: f64,
10}
11
12impl CoordXy {
13    pub fn new(x: f64, y: f64) -> Self {
14        Self { x, y }
15    }
16}
17
18impl From<&CoordXy> for Vec<f64> {
19    fn from(val: &CoordXy) -> Self {
20        vec![val.x, val.y]
21    }
22}
23impl From<CoordXy> for Vec<f64> {
24    fn from(val: CoordXy) -> Self {
25        (&val).into()
26    }
27}
28impl From<Vec<f64>> for CoordXy {
29    fn from(array: Vec<f64>) -> Self {
30        match array.len() {
31            2 => Self {
32                x: array[0],
33                y: array[1],
34            },
35            _ => panic!("Expected array of length 2 or 3, got {}", array.len()),
36        }
37    }
38}
39
40impl Coord for CoordXy {
41    fn has_z() -> bool {
42        false
43    }
44    fn has_m() -> bool {
45        false
46    }
47    fn x(&self) -> f64 {
48        self.x
49    }
50    fn y(&self) -> f64 {
51        self.y
52    }
53    fn from_coord<C: Coord>(c: C) -> Self {
54        Self { x: c.x(), y: c.y() }
55    }
56    fn from_coord_fields(x: f64, y: f64, _z: Option<f64>, _m: Option<f64>) -> Self {
57        Self { x, y }
58    }
59}
